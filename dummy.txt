import socket
def port_scanner(ip, port):
  try:
    with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as s:
      s.settimeout(1)
      s.connect((ip, port))
      print(f"[+] Port {port} is open on {ip}")
  except:
    print(f"[-] Port {port} is closed on {ip}")

ip = "127.0.0.1"
for port in range(45, 50):
  port_scanner(ip, port)

import requests
def check(url, param):
  payload = f"{url}?{param}=' OR '1'='1"
  print(f"Testing: {payload}")
  try:
    response = requests.get(payload, timeout = 5)
    if response.status_code == 500 or "sql" in response.text.lower() or "syntax" in response.text.lower():
      print(f"Potential Vulnerability at {payload}\n")

def caesar_cipher(mesage, shift, mode):
  result = ""
  for char in message:
    if char.isalpha():
      shift_base = 65 if char.isupper() else 97
      shift_value = shift if mode == "encrypt" else -shift
      result += chr((ord(char) - shift_base + shift_value) % 26 + shift_base)
    else:
      result += char
  return result
message = input("Enter a message:\n")
shift = int(input("Enter a shift:\n"))
encrypted = caesar_cipher(message, shift, "encrypt")
print(f"Encryped message: {encrypted}\n")
decrypted = caesar_cipher(message, shift, "decrypt")
